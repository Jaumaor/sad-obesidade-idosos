Buscar Pacientes:
  GET /api/v1/pacientes/buscar
  Query:
    - territorio_ids      : "1,2,3"
    - unidade_saude_ids   : "1,2"
    - idade_minima        : 60
    - idade_maxima        : 80
    - em_acompanhamento   : true/false
    - limite              : 100

Detalhes:
  GET /api/v1/pacientes/{uuid}
  Retorna: dados completos, IMC, PA, glicemia, comorbidades, alertas

Acompanhamentos:
  GET /api/v1/pacientes/{uuid}/acompanhamentos?limite=50
  Retorna: histórico de visitas com medições

Comorbidades:
  GET /api/v1/pacientes/{uuid}/comorbidades
  Retorna: lista de condições crônicas

Alertas:
  GET /api/v1/pacientes/{uuid}/alertas?limite=50
  Retorna: alertas gerados pelo sistema
